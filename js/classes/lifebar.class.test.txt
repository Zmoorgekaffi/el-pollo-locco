class Lifebar {
    color_border = 'orange';
    color_fill = 'red';
    width = 100;
    height = 25;
    x = 50;
    y = 50;
    world;

    constructor() {
        
    }

    draw() {
        this.world.ctx.strokeStyle = this.color_border;
        this.world.ctx.lineWidth = 5;
        this.world.ctx.beginPath();
        this.world.ctx.rect(this.x, this.y, this.width, this.height);
        this.world.ctx.stroke();
        this.world.ctx.fillStyle = this.color_fill;
        this.world.ctx.fill();
    }
}
